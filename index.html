===== Header ===== header = ctk.CTkFrame(app, height=60) header.pack(fill="x", padx=10, pady=10) title = ctk.CTkLabel( header, text="Discord Token Checker • Work: 5 Locked: 0 Invalid: 996", font=("Arial", 16, "bold") ) title.pack(side="left", padx=20) input_btn = ctk.CTkButton(header, text="Input Tokens", width=120) input_btn.pack(side="right", padx=20) # ===== Main Frame ===== main = ctk.CTkFrame(app) main.pack(fill="both", expand=True, padx=10, pady=5) # ===== Left Panel ===== left = ctk.CTkFrame(main, width=220) left.pack(side="left", fill="y", padx=10, pady=10) ctk.CTkLabel(left, text="Outputs", font=("Arial", 14, "bold")).pack(anchor="w", padx=10, pady=5) def checkbox(text): return ctk.CTkCheckBox(left, text=text) checkbox("Work").pack(anchor="w", padx=10, pady=5) checkbox("Locked").pack(anchor="w", padx=10, pady=5) checkbox("Invalid").pack(anchor="w", padx=10, pady=5) ctk.CTkButton(left, text="Custom Filter").pack(fill="x", padx=10, pady=10) ctk.CTkButton(left, text="Advanced Settings").pack(fill="x", padx=10) #
