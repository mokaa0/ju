using UnityEngine; public class MouseCamera : MonoBehaviour { float mouseX; float mouseY; public float sensitivity = 100f; public Transform player; float rotation = 0f; public float minAngle = -90f; public float maxAngle = 90f; void Start() { Cursor.visible = false; Cursor.lockState = CursorLockMode.Locked; } void Update() { mouseX = Input.GetAxis("Mouse X") * sensitivity * Time.deltaTime; mouseY = Input.GetAxis("Mouse Y") * sensitivity * Time.deltaTime; rotation -= mouseY; rotation = Mathf.Clamp(rotation, minAngle, maxAngle); transform.localRotation = Quaternion.Euler(rotation, 0f, 0f); encouraging the player rotation player.Rotate(Vector3.up * mouseX); } }
