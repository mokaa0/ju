// index.js function getRandomInt(min, max) { return Math.floor(Math.random() * (max - min + 1)) + min; } function generateLine() { const words = [ "ban", "all", "kick", "ip", "banip", "ddos", "attack", "exploit", "inject", "packet", "spoof", "override", "access", "root", "admin", "scan", "trojan", "virus", "hack", "crack", "breach", "shell", "botnet", "payload", "Respect", "rk6" ]; let line = []; // نضيف كلمتي Respect و rk6 في كل سطر line.push("Respect"); line.push("rk6"); // نضيف 3 إلى 6 كلمات عشوائية من القائمة بدون Respect و rk6 const filteredWords = words.filter(w => w !== "Respect" && w !== "rk6"); const count = getRandomInt(3, 6); for (let i = 0; i < count; i++) { const word = filteredWords[getRandomInt(0, filteredWords.length - 1)]; line.push(word); } // نخلط ترتيب الكلمات عشوائياً line = line.sort(() => Math.random() - 0.5); return line.join(" "); } function printLines() { let count = 0; const maxLines = 100; // عدد السطور const interval = setInterval(() => { process.stdout.write("\x1b[32m"); // تغيير لون النص للأخضر console.log(generateLine()); count++; if (count >= maxLines) { clearInterval(interval); } }, 10); // 10 مللي ثانية بين السطور = 100 سطر في الثانية تقريباً } printLines();
