const { Client, GatewayIntentBits, Partials, Collection } = require("discord.js") const fs = require("fs") require("dotenv").config() const config = require("./config.json") const client = new Client({ intents: [ GatewayIntentBits.Guilds, GatewayIntentBits.GuildMembers, GatewayIntentBits.GuildMessages, GatewayIntentBits.GuildBans, GatewayIntentBits.MessageContent ], partials: [Partials.Message, Partials.Channel, Partials.Reaction] }) client.commands = new Collection() client.config = config client.saveConfig = () => { fs.writeFileSync("./config.json", JSON.stringify(client.config, null, 2)) } const commandFiles = fs.readdirSync("./commands").filter(file => file.endsWith(".js")) for (const file of commandFiles) { const command = require(`./commands/${file}`) client.commands.set(command.data.name, command) } client.once("ready", () => { console.log(`${client.user.tag} is online`) }) client.on("interactionCreate", async interaction => { if (!interaction.isChatInputCommand()) return const command = client.commands.get(interaction.commandName) if (!command) return try { await command.execute(interaction, client) } catch (err) { console.error(err) await interaction.reply({ content: "An error occurred while executing this command.", ephemeral: true }) } }) require("./securityHandler")(client) client.login(process.env.TOKEN)
